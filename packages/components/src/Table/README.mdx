  {() => {

      { label: 'Color', key: 'color', width: '45%' },

        color: 'red',
        value: '#f00',

        color: 'green',
        value: '#0f0',
      },

      },

      },
    ]

      <Table
        data-test='table'
        rowInstance={rowInstance}

## With Pagination and Sorting

<Playground>
  {() => {
    class StateManager extends React.Component {
      constructor (props, context) {
        super(props, context)
        this.cols = [
          { label: 'Brand', key: 'brand' },
          { label: 'Model', key: 'model' },
          { label: 'Doors', key: 'doors' },
          { label: 'Transmission', key: 'transmission' },
        ]
        this.fetchData = this.fetchData.bind(this)
        this.rowInstance = this.rowInstance.bind(this)
        this.state = {
          data: [],
          paginationProps: {},
          counter: 0,
        }
      }
      rowInstance (row) {
        return (
          <RowInstance key={row.model}>
            <ColumnInstance columnClass='cell'>{row.brand}</ColumnInstance>
            <ColumnInstance columnClass='cell'>{row.model}</ColumnInstance>
            <ColumnInstance columnClass='cell'>{row.doors}</ColumnInstance>
            <ColumnInstance columnClass='cell'>
              {row.transmission}
            </ColumnInstance>
          </RowInstance>
        )
      }
      componentDidMount () {
        this.fetchData(this.state.counter)
      }
      fetchData () {
        const counter = this.state.counter
        const itemsPerPage = 3
        const dataAvailable = [
          {
            brand: 'Nissan',
            model: 'Sentra',
            doors: 4,
            transmission: 'automatic',
          },
          {
            brand: 'Nissan',
            model: 'Maxima',
            doors: 4,
            transmission: 'manual',
          },
          {
            brand: 'Nissan',
            model: 'Skyline',
            doors: 2,
            transmission: 'automatic',
          },
          {
            brand: 'Ford',
            model: 'Taurus',
            doors: 4,
            transmission: 'automatic',
          },
          { brand: 'Ford', model: 'Escort', doors: 4, transmission: 'manual' },
          { brand: 'Audi', model: 'r8', doors: 4, transmission: 'manual' },
          { brand: 'Audi', model: 'q2', doors: 4, transmission: 'automatic' },
          {
            brand: 'Benz',
            model: 'E class',
            doors: 4,
            transmission: 'automatic',
          },
        ]
        const totalCount = dataAvailable.length
        const dataSentCount =
          counter +
          (dataAvailable.length - counter < itemsPerPage
            ? dataAvailable.length - counter
            : itemsPerPage)
        if (dataSentCount <= totalCount) {
          let paginationProps = {
            more: totalCount - dataSentCount,
            count: totalCount,
            fetchData: this.fetchData,
            counter: dataSentCount,
          }
          this.setState(prevState => {
            return {
              counter: dataSentCount,
              data: this.state.data.concat(
                dataAvailable.splice(counter, itemsPerPage),
              ),
              paginationProps: paginationProps,
            }
          })
        }
      }
      render () {
        return (
          <Table
            data-test='paginated-table'
            rowInstance={this.rowInstance}
            sortEnabled={true}
            columns={this.cols}
            data={this.state.data}
            paginationEnabled={true}
            paginationProps={this.state.paginationProps}
          />
        )
      }
    }
    return <StateManager />
  }}
</Playground>
