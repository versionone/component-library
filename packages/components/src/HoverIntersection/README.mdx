# IntersectionHover

Tracks hover state for a component; remove hover state from child intesection regions.

## Basic Example

<Playground>
  {() => {
    const ColorDiv = createComponent(
      ({ hovered }) => ({
        backgroundColor: hovered ? 'lightblue' : 'white',
        padding: 10,
      }),
      'div',
      ['onMouseEnter', 'onMouseLeave'],
    );
    return (
      <HoverIntersection>
        {({ bind, hovered }) => (
          <StyleProvider>
            <ColorDiv {...bind} hovered={hovered}>
              <p>parent div</p>
              <HoverIntersection.Exclude>
                {({ bind, hovered }) => (
                  <ColorDiv {...bind} hovered={hovered}>
                    child div
                  </ColorDiv>
                )}
              </HoverIntersection.Exclude>
            </ColorDiv>
          </StyleProvider>
        )}
      </HoverIntersection>
    );
  }}
</Playground>
