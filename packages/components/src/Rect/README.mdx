import { Rect } from './';

## Usage

Render prop that calls back with the reference to a DOM node and its measurements (from `element.getBoundingClientRect()`).

## Examples

<Playground>
  <Rect>
    {({ ref, rect }) => {
      console.log(JSON.stringify(rect, null, 2));
      return (
        <div>
          <div />
          <div
            ref={ref}
            style={{
              display: 'block',
              padding: 10,
              margin: 5,
              border: '1px solid black',
            }}
          >
            See my measurements in the console log.
          </div>
        </div>
      );
    }}
  </Rect>
</Playground>
